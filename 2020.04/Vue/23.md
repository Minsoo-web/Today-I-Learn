# Vue.js

## 11. 여러개의 Vue 인스턴스 사용하기

```vue
<template>
  <div id="app">
    {{ name }}
    <button @click="changeText">click</button>
  </div>
  <div id="app-1">
    {{ name }}
    <button @click="changeText">click</button>
  </div>
</template>

<script>
const app = new Vue({
  el: "#app",
  data: {
    name: "name",
  },
  methods: {
    changeText() {
      // 서로 다른 vue 인스턴스의 상태에 접근이 가능하다.
      app1.name = "change text";
    },
  },
});
const app1 = new Vue({
  el: "#app-1",
  data: {
    name: "name2",
  },
  methods: {
    changeText() {
      // 서로 다른 vue 인스턴스의 상태에 접근이 가능하다.
      app.name = "change text";
    },
  },
});
</script>
```
